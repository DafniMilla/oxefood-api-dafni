Implementação BIDIRECIONAL

Classe Professor (dono do relacionamento)
@ManyToMany
@JoinTable(
    name = "professor_turma",
    joinColumns = @JoinColumn(name = "professor_id"),
    inverseJoinColumns = @JoinColumn(name = "turma_id")
)
private List<Turma> turmas;
